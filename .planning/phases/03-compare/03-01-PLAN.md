<delta_format_reference>
From RESEARCH.md Pattern 1 â€” jsondiffpatch delta format (verified via Context7 /benjamine/jsondiffpatch):

```typescript
const left  = { a: 1, b: 2, c: 3 };
const right = { a: 99, c: 3, d: 4 };

const delta = diff(left, right);
// delta = {
//   a: [1, 99],          // modified: [oldValue, newValue]              â†’ CHANGED line
//   b: [2, 0, 0],        // deleted: [oldValue, 0, 0]                    â†’ REMOVED line (Left only)
//   d: [4],              // added: [newValue]                            â†’ ADDED line (Right only)
//   // c is absent â€” unchanged
// }
```

Delta entry rules (drive `deltaToLineEntries` and `filterStructural`):
- `[newValue]` (length 1) â†’ ADDED (right side only)
- `[oldValue, newValue]` (length 2) â†’ CHANGED (both sides)
- `[oldValue, 0, 0]` (length 3, value[1]===0 && value[2]===0) â†’ REMOVED (left side only)
- `[unidiff, 0, 2]` (length 3, value[2]===2) â†’ text diff; treat as CHANGED
- `{ _t: 'a', ... }` â†’ array delta; iterate keys (numeric `N` and `_N` for removals)
- Plain object without `_t` â†’ nested object delta; recurse with extended path
- Identical inputs â†’ `diff()` returns `undefined`
</delta_format_reference>

<line_walker_algorithm>
`deltaToLineEntries(delta, formattedJson, side)` walks a delta and produces line entries by matching key paths to lines in the formatted JSON.

Algorithm (V1 â€” sufficient for the typical case; deeply nested duplicate-key edge case documented as known limitation):

1. If `delta === undefined` â†’ return `[]`
2. Split `formattedJson` by `\n` into `lines: string[]`
3. Walk `delta` recursively with a `path: string[]` accumulator (starts `[]`):
   - For each `(key, value)` in `delta`:
     - If `key === '_t'` â†’ skip (marker, not a key)
     - Determine entry type from `value`:
       - `Array.isArray(value) && value.length === 1` â†’ type 'added' (skip on `side === 'left'`)
       - `Array.isArray(value) && value.length === 3 && value[1] === 0 && value[2] === 0` â†’ type 'removed' (skip on `side === 'right'`)
       - `Array.isArray(value) && value.length === 2` â†’ type 'changed' (emit on both sides)
       - `Array.isArray(value) && value.length === 3 && value[2] === 2` â†’ type 'changed' (text diff; emit both sides)
       - `value` is a plain object (typeof 'object', !Array.isArray, !== null) â†’ recurse with `path = [...path, key]`; do NOT emit a line for the parent object itself
   - When emitting a line, call `findLineForPath([...path, key], lines, side)` (see step 4) and push `{ line, type }` if found
4. `findLineForPath(keyPath, lines, side)`:
   - For purely numeric keys (array indices), call `findArrayElementLine(keyPath, lines, side)` (see step 5)
   - Otherwise: walk lines top-down, tracking nesting depth (count `{` and `[` opens minus `}` and `]` closes BEFORE the current line's contents). At each line, the leading whitespace = `depth * 2` spaces (formatted with 2-space indent). A line matches `keyPath[i]` at depth `i` when:
     - The line's leading whitespace equals `i * 2` spaces, AND
     - Trim-left of the line starts with `"` + escapeForJson(keyPath[i]) + `"` followed by `:`
   - Advance `i` from 0 to keyPath.length-1; the matching last line is the result. If `i` cannot fully advance, return `null`.
   - Return the 1-based line number of the final matched line.
5. `findArrayElementLine(keyPath, lines, side)`:
   - Array deltas use keys like `"0"`, `"1"`, `"_0"` (removal). For `_N` keys: only emit on `side === 'left'`. For `N` keys with added/changed: only on side per type rules.
   - Walk to the parent path (excluding the final numeric key); from the parent's matched line, find the corresponding `[` opening, then count `,`-separated top-level elements until index N. Return that line.
   - This is V1 â€” accept the known limitation that arrays containing nested objects with newlines may need refinement; document and ship.
6. Helper `escapeForJson(s)`: returns `s` with `\` â†’ `\\` and `"` â†’ `\"` (matches `JSON.stringify` key encoding).

This algorithm runs entirely on plain strings â€” no CodeMirror view needed â€” so it is fully unit-testable.
</line_walker_algorithm>

    ```typescript
    import { StateField, StateEffect, RangeSetBuilder, type Text, type Extension } from '@codemirror/state';
    import { Decoration, type DecorationSet, EditorView } from '@codemirror/view';

    export type DiffLineType = 'removed' | 'added' | 'changed';
    export interface DiffLineEntry { line: number; type: DiffLineType; }

    // ----- CodeMirror integration -----
    export const setDiffDecorations = StateEffect.define<DecorationSet>();

    export const diffDecorationsField = StateField.define<DecorationSet>({
      create: () => Decoration.none,
      update(deco, tr) {
        deco = deco.map(tr.changes);
        for (const effect of tr.effects) {
          if (effect.is(setDiffDecorations)) deco = effect.value;
        }
        return deco;
      },
      provide: (f) => EditorView.decorations.from(f),
    });

    export const diffTheme: Extension = EditorView.baseTheme({
      '&.cm-editor .cm-diff-removed': { backgroundColor: '#4b1818' },
      '&.cm-editor .cm-diff-added':   { backgroundColor: '#1a3a1a' },
      '&.cm-editor .cm-diff-changed': { backgroundColor: '#3a2a00' },
    });

    // ----- Pure utilities -----
    export function buildPaneDecorations(doc: Text, entries: DiffLineEntry[]): DecorationSet {
      if (entries.length === 0) return Decoration.none;
      const builder = new RangeSetBuilder<Decoration>();
      const sorted = [...entries].sort((a, b) => a.line - b.line);
      for (const { line, type } of sorted) {
        if (line < 1 || line > doc.lines) continue;
        const { from } = doc.line(line);
        const cls = type === 'removed' ? 'cm-diff-removed'
                  : type === 'added'   ? 'cm-diff-added'
                                       : 'cm-diff-changed';
        builder.add(from, from, Decoration.line({ class: cls }));
      }
      return builder.finish();
    }

    function entryType(value: unknown): DiffLineType | null {
      if (!Array.isArray(value)) return null;
      if (value.length === 1) return 'added';
      if (value.length === 2) return 'changed';
      if (value.length === 3 && value[1] === 0 && value[2] === 0) return 'removed';
      if (value.length === 3 && value[2] === 2) return 'changed'; // text diff
      return null;
    }

    export function filterStructural(delta: Record<string, unknown>): Record<string, unknown> {
      const result: Record<string, unknown> = {};
      for (const [key, value] of Object.entries(delta)) {
        if (key === '_t') continue; // re-attach only when array container has surviving entries
        const t = entryType(value);
        if (t === 'added' || t === 'removed') {
          result[key] = value;
        } else if (t === null && typeof value === 'object' && value !== null) {
          const nested = filterStructural(value as Record<string, unknown>);
          if (Object.keys(nested).length > 0) {
            // Preserve _t marker on nested array containers if the source had one
            const sourceMarker = (value as Record<string, unknown>)._t;
            if (sourceMarker !== undefined) (nested as Record<string, unknown>)._t = sourceMarker;
            result[key] = nested;
          }
        }
        // 'changed' (length 2) and text-diff (length 3 + index 2 === 2) â†’ drop
      }
      return result;
    }

    function escapeForJson(s: string): string {
      return s.replace(/\\/g, '\\\\').replace(/"/g, '\\"');
    }

    function findLineForPath(
      keyPath: string[],
      lines: string[],
    ): number | null {
      // Walk lines once, advancing through the keyPath
      let depth = 0;
      let pathIdx = 0;
      let lastMatchedLine = -1;
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];
        // Leading whitespace count (spaces only â€” formatted JSON uses 2-space indent)
        let leading = 0;
        while (leading < line.length && line[leading] === ' ') leading++;
        const trimmed = line.slice(leading);

        if (pathIdx < keyPath.length) {
          const expectedIndent = pathIdx * 2;
          const expectedKey = keyPath[pathIdx];
          // Numeric key (array index) â€” handled by findArrayElementLine path; skip object-key matching
          if (/^_?\d+$/.test(expectedKey)) {
            // Array element matching: need the parent's `[` line to be matched first
            // Simplification for v1: try to match by line offset from parent
            // (See findArrayElementLine below.)
            return null; // signal caller to use array-element path
          }
          if (
            leading === expectedIndent &&
            trimmed.startsWith(`"${escapeForJson(expectedKey)}"`)
          ) {
            lastMatchedLine = i + 1; // 1-based
            pathIdx++;
            if (pathIdx === keyPath.length) return lastMatchedLine;
          }
        }
        // Track structural depth via { [
        for (const ch of line) {
          if (ch === '{' || ch === '[') depth++;
          else if (ch === '}' || ch === ']') depth--;
        }
      }
      return pathIdx === keyPath.length ? lastMatchedLine : null;
    }

    function findArrayElementLine(
      parentPath: string[],
      indexKey: string,    // 'N' for added/changed, '_N' for removed
      lines: string[],
    ): number | null {
      // Strip leading underscore if present (removal markers)
      const idxStr = indexKey.startsWith('_') ? indexKey.slice(1) : indexKey;
      const idx = Number.parseInt(idxStr, 10);
      if (!Number.isFinite(idx)) return null;
      // Find parent line; if parentPath is empty, root array starts at line 1
      const parentLine = parentPath.length === 0 ? 1 : findLineForPath(parentPath, lines);
      if (parentLine === null) return null;
      // From parentLine forward, count top-level array elements at parent's child-depth
      // V1 heuristic: each array element occupies one line at depth (parentDepth + 1) * 2.
      // For an array whose parent is at depth D, array element lines are at indent (D+1)*2.
      // Approximate: from parentLine, return parentLine + idx + 1 as the element's line.
      // This matches the documented V1 array granularity (UI-SPEC: "highlight entire array element lines by index").
      const candidate = parentLine + idx + 1;
      if (candidate < 1 || candidate > lines.length) return null;
      return candidate;
    }

    export function deltaToLineEntries(
      delta: Record<string, unknown> | undefined,
      formattedJson: string,
      side: 'left' | 'right',
    ): DiffLineEntry[] {
      if (!delta) return [];
      const lines = formattedJson.split('\n');
      const out: DiffLineEntry[] = [];

      function walk(node: Record<string, unknown>, path: string[]) {
        const isArrayDelta = node._t === 'a';
        for (const [key, value] of Object.entries(node)) {
          if (key === '_t') continue;

          } else if (typeof value === 'object' && value !== null) {

          }
        }
      }

      walk(delta, []);
      return out;
    }
    ```

    ```typescript
    import { useState, useCallback } from 'react';
    import { diff as jdpDiff } from 'jsondiffpatch';

    export type DiffMode = 'value' | 'structure';

    export interface UseDiffReturn {
      delta: Record<string, unknown> | undefined;
      mode: DiffMode;
      setMode: (m: DiffMode) => void;
      compare: (leftJson: string, rightJson: string) => void;
      reset: () => void;
      diffActive: boolean;
    }

    export function useDiff(): UseDiffReturn {
      const [delta, setDelta] = useState<Record<string, unknown> | undefined>(undefined);
      const [mode, setMode] = useState<DiffMode>('value');
      const [diffActive, setDiffActive] = useState(false);

      // compare() throws SyntaxError on invalid JSON â€” caller surfaces ParseErrorBanner.
      const compare = useCallback((leftJson: string, rightJson: string) => {
        const leftParsed = JSON.parse(leftJson);
        const rightParsed = JSON.parse(rightJson);
        const d = jdpDiff(leftParsed, rightParsed);
        setDelta(d as Record<string, unknown> | undefined);
        setDiffActive(true);
      }, []);

      const reset = useCallback(() => {
        setDelta(undefined);
        setDiffActive(false);
      }, []);

      return { delta, mode, setMode, compare, reset, diffActive };
    }
    ```
