```typescript
// src/workers/jqWorker.ts
import jq from 'jq-web';
import { sanitizeJqError } from '../lib/jqErrors';

type RunMessage = { type: 'run'; expr: string; json: string };

jq.then((jqInst) => {
  self.postMessage({ type: 'ready' });

  self.onmessage = (e: MessageEvent<RunMessage>) => {
    if (e.data.type !== 'run') return;
    const { expr, json } = e.data;
    try {
      const result = jqInst.raw(json, expr);
      self.postMessage({ type: 'result', output: result.trim() });
    } catch (err) {
      self.postMessage({ type: 'error', message: sanitizeJqError(err) });
    }
  };
});
```

From 02-RESEARCH.md `sanitizeJqError` implementation (copy verbatim into src/lib/jqErrors.ts):

```typescript
export function sanitizeJqError(err: unknown): string {
  const msg = err instanceof Error ? err.message : String(err);
  return msg
    .replace(/^Non-zero exit code:\s*\d+\s*/i, '')
    .replace(/^jq:\s*/i, '')
    .split('\n')[0]
    .trim() || 'jq expression failed';
}
```

From 02-RESEARCH.md Pattern 1 — hook implementation (copy verbatim):

```typescript
import { useEffect, useRef, useState, useCallback } from 'react';
import JqWorker from '../workers/jqWorker.ts?worker';

type WorkerMessage =
  | { type: 'ready' }
  | { type: 'result'; output: string }
  | { type: 'error'; message: string };

export function useJqWorker() {
  const workerRef = useRef<Worker | null>(null);
  const [engineReady, setEngineReady] = useState(false);
  const [running, setRunning] = useState(false);
  const [output, setOutput] = useState<string | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const worker = new JqWorker();
    workerRef.current = worker;

    worker.onmessage = (e: MessageEvent<WorkerMessage>) => {
      const msg = e.data;
      if (msg.type === 'ready') {
        setEngineReady(true);
      } else if (msg.type === 'result') {
        setOutput(msg.output);
        setError(null);
        setRunning(false);
      } else if (msg.type === 'error') {
        setError(msg.message);
        setOutput(null);
        setRunning(false);
      }
    };

    return () => {
      worker.terminate();
      workerRef.current = null;
    };
  }, []);

  const run = useCallback((expr: string, json: string) => {
    if (!workerRef.current || !engineReady || running) return;
    setRunning(true);
    setOutput(null);
    setError(null);
    workerRef.current.postMessage({ type: 'run', expr, json });
  }, [engineReady, running]);

  return { engineReady, running, output, error, run };
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Main thread → Web Worker | Structured-clone postMessage; only `{ type: 'run', expr: string, json: string }` is accepted. Expr and json originate from user input. |
| Web Worker → jq-web WASM | expr and json passed as strings into `jqInst.raw(json, expr)`. jq is a sandboxed string-in/string-out evaluator; it cannot touch DOM, network, or the main thread. |
| Worker filesystem → /jq.wasm | Static asset fetched over same-origin HTTPS from Vite-served `public/` directory. No third-party CDN. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01-01 | Tampering | public/jq.wasm (WASM binary supply chain) | mitigate | Pin `jq-web@0.6.2` with `--save-exact` (no `^` or `~`); commit `public/jq.wasm` to the repo so the binary loaded in production is reviewed and byte-identical to the one tested — no runtime CDN fetch. |
| T-02-01-02 | Denial of Service | Worker handling `{ type: 'run' }` | mitigate | Hook sets `running=true` and rejects concurrent `run()` calls until the worker posts result/error. Worker never blocks the main UI thread. Large/expensive expressions degrade the tab but cannot hang the parent app. Accept that jq has no configurable output cap — acceptable for a dev tool. |
| T-02-01-03 | Information Disclosure | Error message leak via sanitizeJqError | mitigate | `sanitizeJqError` strips internal Emscripten prefixes (`Non-zero exit code: N`, `jq:` stack frames) before returning — only the first non-empty line reaches React state. No stack traces or internal paths are exposed to the UI layer. |
| T-02-01-04 | Input Validation (V5) | Worker `onmessage` handler accepts arbitrary payloads from the main thread | mitigate | Handler guards with `if (e.data.type !== 'run') return;` so only the intended message shape is processed. All fields (`expr`, `json`) are used exclusively as jq-web string inputs — never passed to `eval`, `Function`, `innerHTML`, or fetch. |
| T-02-01-05 | Tampering | jq-web synchronous throw from `raw()` | mitigate | try/catch inside `onmessage` — the worker posts `{ type: 'error', message }` and continues listening. Per RESEARCH.md Pitfall 2, `.catch()` would silently swallow the throw. |
| T-02-01-06 | Denial of Service | Worker lifecycle leak on unmount | mitigate | `useEffect` cleanup calls `worker.terminate()` and nulls `workerRef.current`. Verified by Test 9 in useJqWorker.test.ts. |
| T-02-01-07 | Spoofing | Cross-origin worker spawn | accept | Vite `?worker` import bundles the worker as same-origin; browser blocks off-origin Worker URLs by default. No mitigation code needed. |

ASVS L1 coverage: V5 (input validation) is the only applicable category; mitigated by T-02-01-04. V2/V3/V4/V6 do not apply (browser-only, no auth/session/crypto).
